# cmd+k Command Palette & 404 Page Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a keyboard-triggered command palette for quick navigation and a playful terminal-themed 404 page.

**Architecture:** Two independent features. The command palette is a new Astro component included in BaseLayout, using build-time data and client-side vanilla JS. The 404 page is a template-only rewrite of the existing page.

**Tech Stack:** Astro, Tailwind CSS, vanilla JavaScript

## Chunk 2: Command Palette

### Task 2: Create the CommandPalette component

**Files:**

- [ ] **Step 1: Create the component file**

Create `src/components/CommandPalette.astro` with the full implementation:

```astro
---
import { getCollection } from "astro:content";
import { formatDate } from "@lib/notes.ts";

const notes = await getCollection("notes");
const sortedNotes = notes.sort(
  (a, b) => b.data.pubDate.valueOf() - a.data.pubDate.valueOf(),
);

const pages = [
  { title: "Home", href: "/", type: "page" as const },
  { title: "About", href: "/about/", type: "page" as const },
  { title: "Notes", href: "/notes/", type: "page" as const },
];

const noteItems = sortedNotes.map((note) => ({
  title: note.data.title,
  href: `/notes/${note.id}/`,
  type: "note" as const,
  date: formatDate(note.data.pubDate),
}));

const searchItems = [...pages, ...noteItems];
---

<div id="command-palette" class="palette-backdrop" aria-modal="true" role="dialog">
  <div class="palette-modal">
    <div class="palette-input-row">
      <span class="palette-label">⌘K</span>
      <input
        id="palette-input"
        type="text"
        class="palette-input"
        placeholder="Search pages and notes..."
        autocomplete="off"
      />
    </div>
    <div id="palette-results" class="palette-results"></div>
    <div class="palette-footer">
      <span>↑↓ navigate</span>
      <span>↵ open</span>
      <span>esc close</span>
    </div>
  </div>
</div>

<script is:inline set:html={`window.__SEARCH_ITEMS__ = ${JSON.stringify(searchItems)};`} />

<script>
  const items = window.__SEARCH_ITEMS__;
  const backdrop = document.getElementById("command-palette")!;
  const input = document.getElementById("palette-input")! as HTMLInputElement;
  const resultsContainer = document.getElementById("palette-results")!;
  let highlightIndex = 0;
  let currentResults: typeof items = [];

  function open() {
    backdrop.classList.add("visible");
    input.value = "";
    highlightIndex = 0;
    render(items);
    input.focus();
  }

  function close() {
    backdrop.classList.remove("visible");
    input.blur();
  }

  function render(filtered: typeof items) {
    currentResults = filtered;
    const pages = filtered.filter((i: any) => i.type === "page");
    const notes = filtered.filter((i: any) => i.type === "note");

    if (filtered.length === 0) {
      resultsContainer.innerHTML =
        '<div class="palette-empty">No results found</div>';
      return;
    }

    let html = "";

    if (pages.length > 0) {
      html += '<div class="palette-group-label">Pages</div>';
      pages.forEach((item: any, i: number) => {
        const idx = i;
        html += `<div class="palette-item${idx === highlightIndex ? " highlighted" : ""}" data-index="${idx}" data-href="${item.href}">
          <span class="palette-item-title">${item.title}</span>
          <span class="palette-item-meta">${item.href}</span>
        </div>`;
      });
    }

    if (notes.length > 0) {
      html += '<div class="palette-group-label">Notes</div>';
      notes.forEach((item: any, i: number) => {
        const idx = pages.length + i;
        html += `<div class="palette-item${idx === highlightIndex ? " highlighted" : ""}" data-index="${idx}" data-href="${item.href}">
          <span class="palette-item-title">${item.title}</span>
          <span class="palette-item-meta">${item.date || ""}</span>
        </div>`;
      });
    }

    resultsContainer.innerHTML = html;

    resultsContainer.querySelectorAll(".palette-item").forEach((el) => {
      el.addEventListener("click", () => {
        window.location.href = (el as HTMLElement).dataset.href!;
      });
    });
  }

  function updateHighlight(delta: number) {
    if (currentResults.length === 0) return;
    highlightIndex =
      (highlightIndex + delta + currentResults.length) % currentResults.length;
    render(currentResults);

    const highlighted = resultsContainer.querySelector(".highlighted");
    if (highlighted) {
      highlighted.scrollIntoView({ block: "nearest" });
    }
  }

  input.addEventListener("input", () => {
    const query = input.value.toLowerCase();
    const filtered = items.filter((item: any) =>
      item.title.toLowerCase().includes(query),
    );
    highlightIndex = 0;
    render(filtered);
  });

  document.addEventListener("keydown", (e) => {
    const tag = (e.target as HTMLElement)?.tagName;
    const isOpen = backdrop.classList.contains("visible");
    if ((e.metaKey || e.ctrlKey) && e.key === "k") {
      if (!isOpen && (tag === "INPUT" || tag === "TEXTAREA")) return;
      e.preventDefault();
      if (isOpen) {
        close();
      } else {
        open();
      }
      return;
    }

    if (!backdrop.classList.contains("visible")) return;

    if (e.key === "Escape") {
      e.preventDefault();
      close();
    } else if (e.key === "ArrowDown") {
      e.preventDefault();
      updateHighlight(1);
    } else if (e.key === "ArrowUp") {
      e.preventDefault();
      updateHighlight(-1);
    } else if (e.key === "Enter") {
      e.preventDefault();
      if (currentResults.length > 0 && currentResults[highlightIndex]) {
        window.location.href = currentResults[highlightIndex].href;
      }
    } else if (e.key === "Tab") {
      e.preventDefault();
      input.focus();
    }
  });

  backdrop.addEventListener("click", (e) => {
    if (e.target === backdrop) {
      close();
    }
  });
</script>
```
